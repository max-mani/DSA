# STRINGS:

## Easy:

https://leetcode.com/problems/add-strings/description/

```cpp
class Solution {
public:
    string addStrings(string num1, string num2) {
        int i = num1.size() - 1;
        int j = num2.size() - 1;
        int c = 0;
        string out = "";
        while (i >= 0 || j >= 0 || c) {
            int s = c;
            if (i >= 0) {
                s += num1[i] - '0';
                i--;
            }
            if (j >= 0) {
                s += num2[j] - '0';
                j--;
            }
            out += (s % 10) + '0';
            c = s / 10;
        }
        reverse(out.begin(), out.end());
        return out;
    }
};
```
