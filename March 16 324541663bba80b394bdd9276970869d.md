https://leetcode.com/problems/fibonacci-number/description/

```cpp
class Solution {
public:
    int fib(int n) {
        int f0 = 0;
        int f1 = 1;
        for (int i = 0; i < n; i++) {
            int t = f0;
            f0 = f1;
            f1 = f1 + t;
        }
        return f0;
    }
};
```

https://leetcode.com/problems/squares-of-a-sorted-array/description/

```cpp
class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
        int n = nums.size();
        vector<int> out(n);
        
        int l = 0;
        int r = n - 1;
        int p = n - 1;

        while (l <= r) {
            if (abs(nums[l]) > abs(nums[r])) {
                out[p] = nums[l] * nums[l];
                l++;
            } else {
                out[p] = nums[r] * nums[r];
                r--;
            }
            p--;
        }
        return out;
    }
	};
```
